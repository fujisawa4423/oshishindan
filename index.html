  <script>

    const scores = { firebonds: 0, redhopes: 0, iwaki: 0, airybees: 0 };

    let answered = 0;

 

    function addPoints(team) {

      scores[team]++;

      answered++;

      if (answered === 3) showResult();

    }

 

    function showResult() {

      let topTeam = Object.keys(scores).reduce((a, b) => scores[a] > scores[b] ? a : b);

      const teamNames = {

        firebonds: "🏀 福島ファイヤーボンズ",

        redhopes: "⚾ 福島レッドホープス",

        iwaki: "⚽ いわきFC／福島ユナイテッドFC",

        airybees: "🏐 デンソーエアリービーズ"

      };

      document.getElementById("result").innerText = `あなたにおすすめの推しチームは... ${teamNames[topTeam]} です！`;

    }
